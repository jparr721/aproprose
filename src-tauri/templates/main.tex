\newif\if@bubblesent
\newenvironment{textbubble}[2][received]{%
  \def\@bubbletype{#1}%
  \def\@timestamp{#2}%
  \def\@tempa{sent}%
  \ifx\@bubbletype\@tempa%
    \@bubblesenttrue%
  \else%
    \@bubblesentfalse%
  \fi%
  \begin{quote}%
    \ttfamily\small%
    \setlength{\fboxsep}{6pt}%
    \if@bubblesent%
      \raggedleft%
    \fi%
    \begin{lrbox}{\@tempboxa}%
      \begin{minipage}{0.65\linewidth}%
}{%
      \end{minipage}%
    \end{lrbox}%
    \if@bubblesent%
      \colorbox{blue!25}{\usebox{\@tempboxa}}%
    \else%
      \colorbox{gray!10}{\usebox{\@tempboxa}}%
    \fi%
    % Add timestamp below bubble
    \par\vspace{2pt}%
    {\tiny\color{gray}\@timestamp}%
  \end{quote}%
}
